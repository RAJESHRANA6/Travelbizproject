    SqlCommand cmd = new SqlCommand();
    SqlDataReader dr;
    protected void Page_Load(object sender, EventArgs e)
    {

    }
    protected void Button1_Click(object sender, EventArgs e)
    {
        con.Open();
        cmd = new SqlCommand("select * from reg where uname='" + TextBox1.Text + "' and pswd='" + TextBox2.Text + "'", con);//enter username and password to connect to login module
        dr = cmd.ExecuteReader();
        if (dr.Read())
        {
            Session["User"] = TextBox1.Text; //if username is correct then redirect to paytment module page

        }
        else
        {
            Response.Write("<script>alert('Username password Error. pls Try Again Later')</script>"); //enter wrong password
        }
        con.Close();
       
    }
